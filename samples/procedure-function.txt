program exProcedure;
var
   a, b, c,  min, m: integer;
procedure findMin(x, y, z: integer);

begin
   if x < y then
      m:= x;
   else
      m:= y;
   
   if z < m then
      m:= z;
end; { 
   
   end of procedure findMin
   
   }  

begin
   writeln(' Enter three numbers: ');
   readln( a, b, c);
   findMin(a, b, c, min); (* Procedure call *)
   
   writeln(' Minimum: ', min);
end.
