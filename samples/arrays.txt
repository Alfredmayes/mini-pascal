program exArrays;
var
   n: array [1..10] of integer;   (* n is an array of 10 integers *)
   i, j, a: integer;

begin
   (* initialize elements of array n to 0 *)
   i:= 1;
   j:= 1;
   while  j < 10  do     
   begin
       n[ i ] := i + 100;   (* set element at location i to i + 100 *)
       j:= j + 1;
       i:= i + 1;
    (* output each array element's value *)
   end;
   j:=1;
   while j < 10 do
   begin
      writeln('Element[', j, '] = ', n[j] );
      j:= j + 1;
   end;
end.
